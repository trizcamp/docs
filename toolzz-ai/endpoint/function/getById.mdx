### Response Description 
The endpoint returns a response in JSON format with a confirmation response to the request.
<ResponseField name="items" type="array" required>
  List of plugins retrieved.
  <Expandable title="Items">
    <ResponseField name="id" type="string" required>
      Unique identifier for the plugin.
    </ResponseField>
    <ResponseField name="institutionId" type="string" required>
      ID of the associated institution.
    </ResponseField>
    <ResponseField name="name" type="string" required>
      Name of the plugin.
    </ResponseField>
    <ResponseField name="description" type="string" required>
      Description of the plugin.
    </ResponseField>
    <ResponseField name="code" type="string" required>
      Code defining the plugin functionality.
    </ResponseField>
    <ResponseField name="status" type="boolean" required>
      Indicates whether the plugin is enabled.
    </ResponseField>
    <ResponseField name="unityId" type="string" required>
      ID of the associated unity.
    </ResponseField>
    <ResponseField name="userId" type="string" required>
      ID of the user who created or manages the plugin.
    </ResponseField>
    <ResponseField name="iconFileId" type="string">
      Optional ID of the plugin's icon file.
    </ResponseField>
    <ResponseField name="color" type="string" required>
      Color associated with the plugin for UI purposes.
    </ResponseField>
    <ResponseField name="outputs" type="array">
      List of plugin outputs (empty if none exist).
    </ResponseField>
    <ResponseField name="pluginIcon" type="string">
      Optional URL or details of the plugin's icon.
    </ResponseField>
    <ResponseField name="isActivated" type="boolean" required>
      Indicates if the plugin is currently activated.
    </ResponseField>
  </Expandable>
</ResponseField>
<ResponseField name="pluginCount" type="number" required>
  Total number of plugins available.
</ResponseField>
<ResponseField name="activeCount" type="number" required>
  Number of active plugins.
</ResponseField>
<ResponseField name="pluginLimit" type="number" required>
  Maximum number of plugins allowed.
</ResponseField>
