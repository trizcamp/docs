## Response Description 
Below you will see the description of the Get all models response

<ResponseField name="data" type="array" required>
  List of items.
  <Expandable title="properties">
    <ResponseField name="id" type="string" required>
      Unique identifier of the item.
    </ResponseField>
    <ResponseField name="modelName" type="string" required>
      Name of the model associated with the item.
    </ResponseField>
    <ResponseField name="matchPattern" type="string" required>
      Match pattern used for the item.
    </ResponseField>
    <ResponseField name="startDate" type="date" required>
      Start date for the item's configuration.
    </ResponseField>
    <ResponseField name="unit" type="string" required>
      Unit type, e.g., CHARACTERS.
    </ResponseField>
    <ResponseField name="inputPrice" type="integer" required>
      Price for input usage.
    </ResponseField>
    <ResponseField name="outputPrice" type="integer" required>
      Price for output usage.
    </ResponseField>
    <ResponseField name="totalPrice" type="integer" required>
      Total price for the item.
    </ResponseField>
    <ResponseField name="tokenizerId" type="string" required>
      Identifier of the tokenizer.
    </ResponseField>
    <ResponseField name="tokenizerConfig" type="null">
      Configuration of the tokenizer (nullable).
    </ResponseField>
    <ResponseField name="isLangfuseManaged" type="boolean" required>
      Indicates if the item is managed by Langfuse.
    </ResponseField>
  </Expandable>
</ResponseField>
<ResponseField name="meta" type="object" required>
  Metadata for pagination.
  <Expandable title="properties">
    <ResponseField name="page" type="integer" required>
      Current page number.
    </ResponseField>
    <ResponseField name="limit" type="integer" required>
      Limit of items per page.
    </ResponseField>
    <ResponseField name="totalItems" type="integer" required>
      Total number of items available.
    </ResponseField>
    <ResponseField name="totalPages" type="integer" required>
      Total number of pages available.
    </ResponseField>
  </Expandable>
</ResponseField>
