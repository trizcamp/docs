authorization: none
---
<Expandable title="Request">
<ResponseField name="page" type="integer">
  Page number, starts at 1 (nullable).
</ResponseField>
<ResponseField name="limit" type="integer">
  Limit of items per page. Reduce size if API issues occur (nullable).
</ResponseField>
<ResponseField name="userId" type="string">
  Retrieve only scores with this userId associated to the trace (nullable).
</ResponseField>
<ResponseField name="name" type="string">
  Retrieve only scores with this name (nullable).
</ResponseField>
<ResponseField name="fromTimestamp" type="date-time">
  Filter scores created on or after a certain datetime (ISO 8601).
</ResponseField>
<ResponseField name="toTimestamp" type="date-time">
  Filter scores created before a certain datetime (ISO 8601).
</ResponseField>
<ResponseField name="source" type="enum" required>
  Allowed values: ANNOTATION, API, EVAL. Retrieve only scores from a specific source.
</ResponseField>
<ResponseField name="operator" type="string">
  Retrieve only scores with this operator (nullable).
</ResponseField>
<ResponseField name="value" type="double" required>
  Retrieve only scores with this value.
</ResponseField>
<ResponseField name="scoreIds" type="string">
  Comma-separated list of score IDs to limit results (nullable).
</ResponseField>
<ResponseField name="configId" type="string">
  Retrieve only scores with a specific configId (nullable).
</ResponseField>
<ResponseField name="queueId" type="string">
  Retrieve only scores with a specific annotation queueId (nullable).
</ResponseField>
<ResponseField name="dataType" type="enum" required>
  Allowed values: NUMERIC, BOOLEAN, CATEGORICAL. Retrieve only scores with this dataType.
</ResponseField>
<ResponseField name="traceTags" type="array">
  Only scores linked to traces with all these tags will be returned.
  <Expandable title="items">
    <ResponseField name="tag" type="string" required>
      A single trace tag.
    </ResponseField>
  </Expandable>
</ResponseField>
</Expandable>
