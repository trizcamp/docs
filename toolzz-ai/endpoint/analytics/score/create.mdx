<Expandable title="Request">
<ResponseField name="id" type="string" required>
  Unique identifier for the score.
</ResponseField>
<ResponseField name="traceId" type="string" required>
  Trace ID associated with the score.
</ResponseField>
<ResponseField name="name" type="string" required>
  Name of the score (e.g., "novelty").
</ResponseField>
<ResponseField name="value" type="integer" required>
  Value of the score.
</ResponseField>
<ResponseField name="observationId" type="string" required>
  Observation ID associated with the score.
</ResponseField>
<ResponseField name="comment" type="string" required>
  Comment associated with the score.
</ResponseField>
<ResponseField name="dataType" type="string" required>
  Type of data (e.g., NUMERIC, BOOLEAN, CATEGORICAL).
</ResponseField>
<ResponseField name="configId" type="string" required>
  Config ID associated with the score.
</ResponseField>
