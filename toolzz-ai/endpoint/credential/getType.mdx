## Response Description 
Below you will see the description of the Get all Credential by Type response

<ResponseField name="credentials" type="array" required>
  A list of credentials associated with integrations.
  <ResponseField name="id" type="uuid" required>
    Unique identifier for the credential.
  </ResponseField>
  <ResponseField name="name" type="string" required>
    Name of the credential.
  </ResponseField>
  <ResponseField name="userId" type="uuid" required>
    Identifier of the associated user.
  </ResponseField>
  <ResponseField name="unityId" type="uuid" required>
    Identifier of the associated unity.
  </ResponseField>
  <ResponseField name="integrationId" type="uuid" required>
    Identifier of the integration associated with the credential.
  </ResponseField>
  <ResponseField name="secretId" type="uuid" required>
    Secret identifier for the credential.
  </ResponseField>
  <ResponseField name="status" type="boolean" required>
    Indicates whether the credential is active.
  </ResponseField>
  <ResponseField name="virtualKey" type="string" optional>
    Virtual key associated with the credential.
  </ResponseField>
  <ResponseField name="createdAt" type="string" format="date-time" required>
    Date and time when the credential was created.
  </ResponseField>
  <ResponseField name="updatedAt" type="string" format="date-time" optional>
    Date and time when the credential was last updated.
  </ResponseField>
</ResponseField>

<ResponseField name="integration" type="array" required>
  Details about the integration associated with the credential.
  <ResponseField name="id" type="uuid" required>
    Unique identifier for the integration.
  </ResponseField>
  <ResponseField name="name" type="string" required>
    Name of the integration.
  </ResponseField>
  <ResponseField name="iconId" type="uuid" required>

  </ResponseField>
  <ResponseField name="type" type="string" required>
    Type of integration (e.g., llm).
  </ResponseField>
  <ResponseField name="status" type="boolean" required>
    Indicates whether the integration is active.
  </ResponseField>
  <ResponseField name="validationEndpoint" type="string" optional>
    URL of the validation endpoint for the integration.
  </ResponseField>
  <ResponseField name="slug" type="string" optional>
    Slug of the integration.
  </ResponseField>
  <ResponseField name="form" type="array" required>
    Form fields for integration configuration.
    <ResponseField name="name" type="string" required>
      Name of the form field.
    </ResponseField>
    <ResponseField name="type" type="string" required>
      Type of the form field (e.g., string).
    </ResponseField>
    <ResponseField name="label" type="string" required>
      Label of the form field.
    </ResponseField>
    <ResponseField name="maxLenght" type="number" optional>
      Maximum length of the form field value.
    </ResponseField>
    <ResponseField name="required" type="boolean" required>
      Indicates if the field is mandatory.
    </ResponseField>
  </ResponseField>
  <ResponseField name="createdAt" type="string" format="date-time" required>
    Date and time when the integration was created.
  </ResponseField>
  <ResponseField name="updatedAt" type="string" format="date-time" optional>
    Date and time when the integration was last updated.
  </ResponseField>
  <ResponseField name="icon" type="object" optional>

    <ResponseField name="id" type="uuid" required>
      Unique identifier for the icon.

    <ResponseField name="createdAt" type="string" format="date-time" required>

    <ResponseField name="url" type="string" required>

    <ResponseField name="extension" type="string" required>

    <ResponseField name="maskName" type="string" optional>
      Mask name associated with the icon.
    </ResponseField>
  </ResponseField>
