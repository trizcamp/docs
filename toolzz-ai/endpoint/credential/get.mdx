## Response Description 
Below you will see the description of the Get all Credential in Unity response

<ResponseField name="items" type="array">
  List of credentials and their associated integrations.
  <Expandable title="Credential properties">
    <ResponseField name="id" type="string" required>
      Unique identifier for the credential.
    </ResponseField>
    <ResponseField name="name" type="string" required>
      Name of the credential.
    </ResponseField>
    <ResponseField name="userId" type="string" required>
      User ID associated with the credential.
    </ResponseField>
    <ResponseField name="unityId" type="string" required>
      Unity ID associated with the credential.
    </ResponseField>
    <ResponseField name="integrationId" type="string" required>
      Integration ID for the credential.
    </ResponseField>
    <ResponseField name="secretId" type="string" required>
      Secret ID for the credential.
    </ResponseField>
    <ResponseField name="status" type="boolean" required>
      Indicates if the credential is active.
    </ResponseField>
    <ResponseField name="virtualKey" type="string">
      Virtual key for the credential (optional).
    </ResponseField>
    <ResponseField name="credentials" type="object">
      Additional nested credentials (if any).
    </ResponseField>
    <ResponseField name="createdAt" type="string" required>
      Creation timestamp of the credential.
    </ResponseField>
    <ResponseField name="updatedAt" type="string">
      Last updated timestamp of the credential.
    </ResponseField>
    <ResponseField name="integration" type="object">
      Associated integration details.
      <Expandable title="Integration properties">
        <ResponseField name="id" type="string" required>
          Unique identifier of the integration.
        </ResponseField>
        <ResponseField name="name" type="string" required>
          Name of the integration.
        </ResponseField>
        <ResponseField name="iconId" type="string">
          ID of the associated icon (optional).
        </ResponseField>
        <ResponseField name="type" type="string" required>
          Type of the integration (e.g., monetization, analytics).
        </ResponseField>
        <ResponseField name="status" type="boolean" required>
          Indicates if the integration is active.
        </ResponseField>
        <ResponseField name="validationEndpoint" type="string">
          Endpoint for validation (optional).
        </ResponseField>
        <ResponseField name="slug" type="string">
          Slug for the integration (optional).
        </ResponseField>
        <ResponseField name="form" type="array">
          Form fields required for the integration.
          <Expandable title="Form properties">
            <ResponseField name="name" type="string" required>
              Name of the form field.
            </ResponseField>
            <ResponseField name="label" type="string" required>
              Label for the form field.
            </ResponseField>
            <ResponseField name="type" type="string" required>
              Type of the form field (e.g., text, string).
            </ResponseField>
            <ResponseField name="maxLength" type="number">
              Maximum allowed length for the field.
            </ResponseField>
            <ResponseField name="required" type="boolean" required>
              Indicates if the field is mandatory.
            </ResponseField>
            <ResponseField name="description" type="string">
              Description of the form field (optional).
            </ResponseField>
          </Expandable>
        </ResponseField>
        <ResponseField name="createdAt" type="string" required>
          Creation timestamp of the integration.
        </ResponseField>
        <ResponseField name="updatedAt" type="string">
          Last updated timestamp of the integration (optional).
        </ResponseField>
        <ResponseField name="icon" type="object">
          Icon associated with the integration.
          <Expandable title="Icon properties">
            <ResponseField name="id" type="string" required>
              Unique identifier for the icon.
            </ResponseField>
            <ResponseField name="createdAt" type="string" required>
              Creation timestamp of the icon.
            </ResponseField>
            <ResponseField name="updatedAt" type="string">
              Last updated timestamp of the icon (optional).
            </ResponseField>
            <ResponseField name="institutionId" type="string" required>
              Institution ID associated with the icon.
            </ResponseField>
            <ResponseField name="type" type="string" required>
              Type of the icon (e.g., WidgetAvatarIcon).
            </ResponseField>
            <ResponseField name="fileName" type="string" required>
              Name of the file.
            </ResponseField>
            <ResponseField name="internalStorageUrl" type="string" required>
              Internal storage URL for the icon.
            </ResponseField>
            <ResponseField name="url" type="string" required>
              Public URL for the icon.
            </ResponseField>
            <ResponseField name="size" type="number" required>
              Size of the icon file in bytes.
            </ResponseField>
            <ResponseField name="extension" type="string" required>
              File extension of the icon (e.g., png).
            </ResponseField>
            <ResponseField name="maskName" type="string" required>
              Mask name for the icon.
            </ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
